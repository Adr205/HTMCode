
function fibonacci(n:int): int {
    var a: int = 0;
    var b: int = 1;
    var c: int = 0;

    print(a);
    print(b);
    for(i = 1; i++; i < n-1) {
        c = a + b;
        a = b;
        b = c;
        print(c);
    }
    return c;
}



main(){
    var result : int;
    result = fibonacci(10);
    print("Result", result);
}
